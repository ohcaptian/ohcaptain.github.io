title:链表
#include
using namespace std;

#define ERROR 0;
#define OK 1;

typedef int Elemtype;
typedef int Status;

//单链表结构定义
typedef struct Lnode
{
Elemtype data;
struct Lnode next;
}Lnode,linklist;

//有头节点存在时初始化操作
Status initlist(linklist &L)//L就是头指针，头指针指向头节点
{
L = new Lnode;

L->next=NULL;//L->next就是指向首元节点的位置，首元节点=null，即链表为空

}

Status destroylist(linklist &L)
{
linklist p;
while(L)
{
p=L;
L=L->next;
delete p;

}

}

Status listlength(linklist &L)
{
int i= 0;
linklist p;
p=L->next;
while(p)
{
++i;
p->next;
}
return i;
}

//查找数据元素的节点；
Lnode *LocateELem_L(linklist &L,Elemtype e)
{
linklist p;
p=L->next;//p指向首元节点
while(p&&p->data!=e)
p=p->next;
return p;

}

Status LocateElem(linklist &L ,Elemtype e )
{
int i=0;
linklist p;
p = L->next;
while (p)
{
++i;
if(p->data==e)
return i;
p=p->next;
}

}

//生成一个链表从尾部插入节点
Status createlist_tail(linklist &L ,int n)
{
linklist q,p;
L=new Lnode;
L->next=NULL;//初始化L链表；
p=L;
for(int i=0;i<n;++i)
{
q=new Lnode;//创建一个节点准备插入
cin>>q->data;
q->next=NULL;//尾插法，这是最后一个节点；
p->next=q;
p=q;//p指向新的尾节点；
}
return 0;
}

//生成一个链表从头部插入节点
Status createlist_head(linklist &L ,int n)
{
linklist p,q;
L=new Lnode;
L->next=NULL;
for(int i=0;i<n;i++)
{
p=new Lnode;
cin>>p->data;
p->next=L->next;//p->next 指向原来L->next指向的地方
L->next=p;
}
}

//将一个数据插入链表中；
Status ListInsert(linklist &L,int i,Elemtype e)
{
int j;
linklist p,s;
p = L;
j = 0;
while (p&&j<i-1) /* 寻找第i-1个结点/
{
p = p->next;
++j;
}
if (!p || j > i-1)
return ERROR; / 第i个元素不存在 */

s=new Lnode;
s->data = e;
s->next = p->next;      /* s->next这个指针指向p->next这个节点 */
p->next = s;          /* p->next指向s这个节点  */
return OK;

}

//求两者的交集；
Status Insertset(linklist &a,linklist&b,linklist &c)
{
linklist p = a->next;
int i = 0;
while(p)
{
if(LocateElem(b,p->data))//在b链表中寻找有没有p->data这个数据，如果有插c表中；
ListInsert(c,++i,p->data);
p = p->next;
}
}
//求两者的差集
Status subsset(linklist &a,linklist&b,linklist &c)
{
initlist(c);
linklist p = a->next;
int i = 0;
while(p)
{
if(!LocateElem(b,p->data))
ListInsert(c,1,p->data);
p = p->next;
}
}
//遍历整个链表

linklist p;
p=L->next;

int main() {
linklist L,M,c;

createlist_tail(L,4);

createlist_head(M,4);
Travle(L);
Travle(M);
cout<<”L-M的差集”<<endl;
